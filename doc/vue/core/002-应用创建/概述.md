```js
import { createApp } from 'vue'
// SFC会在构建的时候被转换为组件配置对象
import App from './components/App.vue'

// 创建应用实例 「 可以存在多个，虽然可能没有必要 」
const app = createApp({
  /* 根组件选项 */
})

// 所以createApp也可以直接传入组件
const app1 = createApp(<App />)
```

应用是组件树，每个应用都需要一个“根组件”，其他组件将作为其子组件

所以`createApp`传入的本质就是根组件



```js
import { createApp } from 'vue'
import App from './components/App.vue'

const app = createApp(<App />)

// app是应用实例 -- 进行全局配置
app.component('TodoDeleteButton', TodoDeleteButton)

// 挂载应用
// 1. 挂载应用本质就是渲染应用
// 2. 参数是 DOM 元素或CSS字符串 -- 用于查找挂载元素
// 3. 应用会替换挂载点下原本内容 「 挂载元素本身不属于应用的一部分 」
// 4. mount方法返回的是 根组件实例而非应用实例
app.mount('#app')                       
```
