```ts
type A = { x:number, y?:number }; // 通过?表示可选属性
type B = { x:number, y:number|undefined };

const ObjA:A = { x: 1 }; // 正确 -- undefined 可以省略，也可以显示设置
const ObjB:B = { x: 1 }; // 报错 -- undefined 必须显示设置

const ObjAA:A = { x: 1, y: undefined }; // 正确
const ObjBB:B = { x: 1, y: undefined }; // 正确
```
