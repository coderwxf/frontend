```ts
import { createWebHistory, createRouter } from 'vue-router'
import HelloVue from '/src/views/HelloVue.vue'
import HelloWorld from '/src/views/HelloVue.vue'
import NotFound from '/src/views/HelloVue.vue'

export default createRouter({
  history: createWebHistory(),
  routes: [
    {
      path: '/',
      redirect: '/hello-vue'
    },

    {
      path: '/hello-vue',
      component: HelloVue
    },

    {
      path: '/hello-world',
      component: HelloWorld
    },

    {
      path: '/:pathMatch(.*)',
      component: NotFound
    }
  ]
})
```
