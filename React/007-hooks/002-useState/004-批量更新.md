## 批量更新

```jsx
import { useState } from 'react'

export default function App(props) {
  // useState可以和类组件的state一样存储所有状态
  const [point, setPoint] = useState({
    x: 0,
    y: 0
  })

  function changeY() {
    setPoint({
      ...point // 状态更新函数并不支持部分更新, 需要先浅拷贝之前的数据，所以不推荐使用
      y: point.y + 10
    })
  }

  return (
    <>
      <div>{ point.x } - { point.y }</div>
      <button onClick={changeY}>change Y</button>
    </>
  )
}
```
