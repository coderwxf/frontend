`使用`

```jsx
import Dialog from './Dialog'

export default function App(props) {
  return (
    <>
      <Dialog title="title1" content="content1" />

      <br />

      <Dialog title="title2" content="content2">
        <button>confirm</button>
        <button>cancel</button>
      </Dialog>

      <br />

      <Dialog />
    </>
  )
}
```



`定义`

```jsx
import React from 'react'
import PropTypes from 'prop-types'

function Dialog(props) {
  const { title, content } = props
  const children = React.Children.toArray(props.children)

  return (
    <>
      <h1>{title}</h1>
      <p>{content}</p>
      {
        children.length ?
        (
          <div>
            {children}
          </div>
        ) : null
      }
    </>
  )
}

Dialog.propTypes = {
  title: PropTypes.string,
  content: PropTypes.string,
  // PropTypes.node 表示任何可以被react渲染的东西
  // 包括数字、字符串、元素或者包含这些类型的数组
  // 所以 PropTypes.node 一般可以作为 props.children 的接收类型
  children: PropTypes.node,
}

Dialog.defaultProps = {
  title: 'default title',
  content: 'default content',
  children: <button>default button</button>
}

export default Dialog
```
