1. 一般不推荐使用非受控组件

2. 非受控组件使用`defaultValue/defaultChecked`来绑定默认值
3. 非受控组件就是状态直接交给表单元素自己管理，不交给react进行管理

```jsx
import { PureComponent, createRef } from 'react'

export default class App extends PureComponent {
  state = {
    name: 'React',
    isAgree: false
  }

  checkboxRef = createRef()
  inputRef = createRef()

  getInfo = () => {
    console.log(this.inputRef.current.value)
    console.log(this.checkboxRef.current.checked)
  }

  render() {
    return (
      <>
        <input type="text" defaultValue={this.state.name} ref={this.inputRef} />
        <input
          type="checkbox"
          defaultChecked={this.state.isAgree}
          ref={this.checkboxRef}
        /> isAgree
        <button onClick={this.getInfo}>getInfo</button>
      </>
    )
  }
}
```
