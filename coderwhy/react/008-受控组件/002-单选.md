```jsx
import { PureComponent } from 'react'

export default class App extends PureComponent {
  state = {
    info: {
      name: '',
      password: '',
      isAgree: false,
      course: ''
    },

    courses: [
      { name: 'math', isChecked: false },
      { name: 'chinese', isChecked: false },
      { name: 'english', isChecked: false },
    ]
  }

  handleSingleInputChange(e) {
    this.setState({
      info: {
        ...this.state.info,
        [e.target.name]: e.target.type === 'checkbox' ? e.target.checked : e.target.value
      }
    })
  }

  getInfo() {
    console.log(this.state.info)
  }

  render() {
    return (
      <>
        {/* 普通input输入框 */}
        <div>
          <label htmlFor="name">
            name:&nbsp;
            <input
              id="name"
              name="name"
              type="text"
              value={this.state.info.name}
              onChange={e => this.handleSingleInputChange(e)}
            />
          </label>
        </div>
        <br />
        {/* 普通input输入框 */}
        <div>
          <label htmlFor="password">
            password:&nbsp;
            <input
              id='password'
              name="password"
              type="text"
              value={this.state.info.password}
              onChange={e => this.handleSingleInputChange(e)}
            />
          </label>
        </div>
        <br />
        {/* 单选框 */}
        <div>
          <label htmlFor="isAgree">
            <input
              id='isAgree'
              name="isAgree"
              type="checkbox"
              value={this.state.info.isAgree}
              onChange={e => this.handleSingleInputChange(e)}
            />&nbsp;
            agree
          </label>
        </div>
        <br />
        {/* 下拉列表 -- 单选 */}
        <div>
          <label htmlFor='course'>
            courses:&nbsp;
            <select id='course' name='course' value={this.state.info.course} onChange={ e => this.handleSingleInputChange(e) }>
              {
                this.state.courses.map(course => (
                  <option key={course.name} value={course.name}>{course.name}</option>
                ))
              }
            </select>
          </label>
        </div>
        <br />
        <button onClick={() => this.getInfo()}>getInfo</button>
      </>
    )
  }
}
```
