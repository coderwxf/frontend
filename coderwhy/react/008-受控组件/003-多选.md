```jsx
import { PureComponent } from 'react'

export default class App extends PureComponent {
  state = {
    info: {
      hobbies: [],
      courses: []
    },

    courses: [
      { name: 'math', isChecked: false },
      { name: 'chinese', isChecked: false },
      { name: 'english', isChecked: false },
    ],

    hobbies: [
      { name: 'football', isChecked: false },
      { name: 'basketball', isChecked: false },
      { name: 'swim', isChecked: false }
    ]
  }

  multipleSelectChange(e) {
    this.setState({
      info: {
        ...this.state.info,
        // 1. e.target.selectedOptions 是一个 HTMLCollection 「 伪数组对象 」，包含了被选中的 <option> 元素
        // 2. Array.from(可迭代对象，mapFn，thisArg) 方法从一个类似数组或可迭代对象创建一个新的，浅拷贝的数组实例
        courses: Array.from(e.target.selectedOptions).map(option => option.value)
      }

    })
  }

  // 多选需要自己添加isChecked属性, 并进行过滤
  selecHobbies(index) {
    const hobbies = [...this.state.hobbies]
    hobbies[index].isChecked = !hobbies[index].isChecked
    this.setState({
      info: {
        ...this.state.info,
        hobbies: hobbies.filter(hobby => hobby.isChecked).map(hobby => hobby.name)
      }
    })
  }

  getInfo() {
    console.log(this.state.info)
  }

  render() {
    return (
      <>
        {/* 多选框 */}
        {
          this.state.hobbies.map((hobby, index) => (
            <label key={hobby.name} htmlFor={hobby.name}>
              <input type="checkbox" id={hobby.name} value={hobby.isChecked} onChange={() => this.selecHobbies(index)} /> { hobby.name }
            </label>
          ))
        }
        <br />
        {/* 下拉列表 --- 多选 */}
        <select multiple value={this.state.info.courses} onChange={e => this.multipleSelectChange(e)}>
          {
            this.state.courses.map(course => (
              <option key={course.name}>{ course.name }</option>
            ))
          }
        </select>
        <br />
        <button onClick={() => this.getInfo()}>getInfo</button>
      </>
    )
  }
}
```
