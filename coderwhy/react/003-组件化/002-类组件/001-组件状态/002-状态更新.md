```jsx
import { Component } from 'react'

export default class App extends Component {
  state = {
    name: 'Klaus',
    age: 23
  }

  changeName() {
    // 1. 状态更新 --- 支持部分状态更新 -- 底层使用Object.assign方法
    // 2. 触发render方法，重新渲染对应组件
    this.setState({
      name: 'Alex'
    })
  }

  render() {
    return (
      <>
        <div>{ this.state.name }</div>
        <button onClick={() => { this.changeName() }}>change name</button>
      </>
    )
  }
}
```
