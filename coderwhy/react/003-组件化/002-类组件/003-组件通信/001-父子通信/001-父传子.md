`父组件`

```jsx
import React, { Component } from 'react'
// 脚手架导入 可以省略JSX后缀
import Child from './Child'

export class App extends Component {
  render() {
    return (
      {/* age="23"会导致传入的props变成字符串，需要使用小括号包裹 */}
      <Child name='Klaus' age={23} />
    )
  }
}

export default App
```



`子组件`

```jsx
import React, { Component } from 'react'
// 导入类型校对工具类
import PropTypes from 'prop-types'

export class Child extends Component {
  // 默认构造器可以省略
  // constructor(props) {
  // 	super(props)
  // }
  
  // 定义属性校对规则 --- 需要作为组件的静态属性使用
  static propTypes = {
    name: PropTypes.string.isRequired, // 字符串必传
    age: PropTypes.number.isRequired,
    gender: PropTypes.oneOf(['male', 'female']) // 值是两个值中的一个
  }
  /*
    类似校验规则查看prop-types官网
    + PropTypes.func -- 函数
    + PropTypes.bool -- 布尔
    + PropTypes.node -- node节点 包括文本，注释，元素等
    + PropTypes.element -- JSX元素
    + PropTypes.elementType -- 自定义组件
  */

  // 设置默认值 --- 需要作为组件的静态属性使用
  static defaultProps = {
    gender: 'male'
  }

  render() {
    const { name, age, gender } = this.props

    return (
      <>
        <div>{ name }</div>
        <div>{ age }</div>
        <div>{ gender }</div>
      </>
    )
  }
}

export default Child
```



## spread props

```jsx
{/* 下述两种写法是等价的 */}
<Child name='Klaus' age={23} />

{/* 将对象展开并依次传入 -- spread props */}
<Child {...this.state.user} />
```
